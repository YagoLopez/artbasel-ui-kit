# Footer Component

## Usage

```jsx
<Footer
  footerData={footerData}
  socialData={socialData}
  paymentData={paymentData}
  linkRenderer={(link, children, target) => <a href={link} target={target}>{children}</a>}
  legalData={legalData}
/>
```

### 4. linkRenderer Samples

As the Footer component can be consumed in several diferent systems, with diferent navigation methods, the links must be rendered in different ways. To add flexibility, the **linkRenderer** property will be used to render all links.

**IMPORTANT**: To have the correct look and fell, the label must be inside an "a tag".

#### Example 1 - "a" tag

```jsx
<Footer
  ...
  linkRenderer={(link, children, target) => <a href={link} target={target}>{label}</a>}
  ...
/>
```


#### Example 2 - Link from router

```jsx
import Link from 'next/link'

<Footer
  ...
  linkRenderer={(link, label, target) =>
    <Link href={link}>
        <a target={`${target === '_blank' ? '_blank' : '_self'}`}>{label}</a>
    </Link>
  }
  ...
/>
```

### 6. footerData JSON Validation Schema

This is the JSON schema used by Footer to validate the footerData before render the content.

```json
{
  "definitions": {},
  "title": "Root",
  "type": "object",
  "required": [
    "cols"
  ],
  "properties": {
    "cols": {
      "$id": "#root/cols",
      "title": "Cols",
      "type": "array",
      "default": [],
      "items": {
        "$id": "#root/cols/items",
        "title": "Items",
        "type": "array",
        "default": [],
        "items": {
          "$id": "#root/cols/items/items",
          "title": "Items",
          "type": "object",
          "required": [
            "title",
            "data"
          ],
          "properties": {
            "title": {
              "$id": "#root/cols/items/items/title",
              "title": "Title",
              "type": "string",
              "default": "",
              "examples": [
                "Buy artworks"
              ],
              "pattern": "^.*$"
            },
            "data": {
              "$id": "#root/cols/items/items/data",
              "title": "Data",
              "type": "array",
              "default": [],
              "items": {
                "$id": "#root/cols/items/items/data/items",
                "title": "Items",
                "type": "object",
                "required": [
                  "label",
                  "link"
                ],
                "properties": {
                  "label": {
                    "$id": "#root/cols/items/items/data/items/label",
                    "title": "Label",
                    "type": "string",
                    "default": "",
                    "examples": [
                      "All artworks"
                    ],
                    "pattern": "^.*$"
                  },
                  "link": {
                    "$id": "#root/cols/items/items/data/items/link",
                    "title": "Link",
                    "type": "string",
                    "default": "",
                    "examples": [
                      "#"
                    ],
                    "pattern": "^.*$"
                  },
                  "target": {
                    "$id": "#root/cols/items/items/data/items/target",
                    "title": "Target",
                    "type": "string",
                    "default": "",
                    "examples": [
                      "_blank"
                    ],
                    "pattern": "^.*$"
                  }
                }
              }
            }
          }
        }
      }
    }
  }
}
```
