import { Meta, Story, Canvas, ArgsTable, Preview } from '@storybook/addon-docs';

import { Container, Row, Col } from '../../Grid';
import SpecialRoomCard from './SpecialRoomCard';
import { example, imagesArray } from './SpecialRoomCard.example';

<Meta
  title="Components/Structure/cards/Special Room Card"
  component={SpecialRoomCard}
  decorators={[withDesign]}
/>

export const Template = (args) => <SpecialRoomCard {...args} />;

export const TemplateList = (args) => {
  return (
    <Container>
      <Row>
        {Array(6)
          .fill(example)
          .map((card, i) => {
            return (
              <Col
                key={`col-${i}`}
                xs={12}
                sm={12}
                md={6}
                lg={4}
                xl={4}
                xxl={4}
                className="mb-lg-9 mb-md-9"
              >
                <SpecialRoomCard {...{ ...args, imageUrl: imagesArray[i] }} />
              </Col>
            );
          })}
      </Row>
    </Container>
  );
};

# Special Room Card

These type of cards make up most of the interface and are key components for logically grouping page content and making it scalable.
